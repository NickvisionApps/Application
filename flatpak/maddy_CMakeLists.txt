cmake_minimum_required(VERSION 3.12 FATAL_ERROR)

project(maddy LANGUAGES CXX)

include(GNUInstallDirs)

add_library(maddy INTERFACE)
target_include_directories(maddy INTERFACE $<INSTALL_INTERFACE:${CMAKE_INSTALL_INCLUDEDIR}>)

install(TARGETS maddy EXPORT maddy)
install(EXPORT maddy FILE maddy-config.cmake DESTINATION "${CMAKE_INSTALL_DATAROOTDIR}/maddy" NAMESPACE maddy::)
install(DIRECTORY "${CMAKE_SOURCE_DIR}/include/maddy" DESTINATION "${CMAKE_INSTALL_INCLUDEDIR}")
